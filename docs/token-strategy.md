### 6.3 Voting Power Calculation

```typescript
function calculateVotingPower(params: {
  stakedAmount: bigint,
  lockDuration: number,     // days
  delegatedAmount: bigint,
  reputationScore: number   // 0-100
}): bigint {
  // Base voting power = staked amount
  let power = params.stakedAmount;

  // Lock duration multiplier (up to 2x for 365 days)
  const lockMultiplier = Math.min(2, 1 + (params.lockDuration / 365));
  power = BigInt(Math.floor(Number(power) * lockMultiplier));

  // Add delegated power (1:1)
  power += params.delegatedAmount;

  // Reputation bonus (up to 50% for 100 reputation)
  const reputationBonus = params.reputationScore / 200; // 0 to 0.5
  power = BigInt(Math.floor(Number(power) * (1 + reputationBonus)));

  return power;
}
```

## 8. Tokenomics Simulation

Quantitative models to validate sustainability and optimize parameters.

### 8.1 Supply Dynamics Model

```typescript
interface SupplySimulation {
  // Initial state
  initialSupply: 1_000_000_000,
  initialCirculating: 130_000_000,  // 13% at launch

  // Emission schedule
  yearlyEmissions: [
    100_000_000,  // Year 1: 100M
    75_000_000,   // Year 2: 75M
    50_000_000,   // Year 3: 50M
    25_000_000,   // Year 4: 25M
  ],

  // Burn mechanisms
  estimatedAnnualBurn: 0.02,  // 2% of circulating from fees

  // Staking assumptions
  estimatedStakingRatio: 0.60,  // 60% of circulating staked
}

function simulateSupply(years: number): SupplyProjection {
  let circulating = 130_000_000;
  let totalBurned = 0;

  const projections = [];

  for (let year = 1; year <= years; year++) {
    // Add emissions
    const emission = yearlyEmissions[Math.min(year - 1, 3)] || 25_000_000;
    circulating += emission;

    // Subtract burns
    const burned = circulating * estimatedAnnualBurn;
    circulating -= burned;
    totalBurned += burned;

    projections.push({
      year,
      circulating,
      staked: circulating * estimatedStakingRatio,
      liquid: circulating * (1 - estimatedStakingRatio),
      totalBurned
    });
  }

  return projections;
}
```
