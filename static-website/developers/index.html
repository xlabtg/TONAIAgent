api-example.sh 
# Create a new agent
curl -X POST https://api.tonaiagent.com/v1/agents \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "My Trading Agent",
    "strategy": "momentum",
    "config": {
      "risk_level": "medium",
      "max_position_size": 1000
    }
  }'

# Response
{
  "id": "agent_abc123",
  "status": "active",
  "created_at": "2024-01-15T10:30:00Z"
}
